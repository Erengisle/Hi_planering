function buildMenu() { const menu = document.getElementById('menu'); if (!menu) return; menu.innerHTML = ''; // Rensa menyn const moments = [...new Set(data.map(row => row.moment))]; moments.forEach(moment => { if (moment === 'Omprov') return; const li = document.createElement('li'); const a = document.createElement('a'); a.href = '#'; a.dataset.section = moment.toLowerCase().replace(/\s+/g, '-'); a.textContent = moment; a.addEventListener('click', (e) => { e.preventDefault(); showSection(a.dataset.section); document.querySelectorAll('nav a').forEach(l => l.classList.remove('active')); a.classList.add('active'); }); li.appendChild(a); menu.appendChild(li); }); const reexamLink = document.querySelector('.reexam-link'); if (reexamLink) { reexamLink.addEventListener('click', (e) => { e.preventDefault(); showSection('reexam'); document.querySelectorAll('nav a').forEach(l => l.classList.remove('active')); reexamLink.classList.add('active'); }); } } function buildSections() { const main = document.querySelector('main'); if (!main) return; // Ta bort tidigare dynamiska sektioner (men behåll intro och reexam) const oldSections = main.querySelectorAll('.section:not(#intro):not(#reexam)'); oldSections.forEach(s => s.remove()); const moments = [...new Set(data.map(row => row.moment))]; moments.forEach(moment => { if (moment === 'Omprov') return; const section = document.createElement('div'); section.id = moment.toLowerCase().replace(/\s+/g, '-'); section.className = 'section'; const h1 = document.createElement('h1'); h1.textContent = moment; section.appendChild(h1); const table = document.createElement('table'); table.innerHTML = `VeckaDag 1Dag 2Kommentar`; const rows = data.filter(row => row.moment === moment); rows.forEach(row => { const tr = document.createElement('tr'); tr.innerHTML = `${row.vecka}${row.dag1}${row.dag2}${row.kommentar}`; table.appendChild(tr); }); section.appendChild(table); main.appendChild(section); }); } function showSection(id) { document.querySelectorAll('.section').forEach(s => s.classList.remove('active')); const target = document.getElementById(id); if (target) target.classList.add('active'); if (window.innerWidth <= 768) document.querySelector('nav').classList.remove('open'); } // Initiera när sidan laddas loadData();
